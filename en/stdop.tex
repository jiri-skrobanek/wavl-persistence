\begin{algorithm}
	\small
	\SetAlgoLined
	\KwResult{$v$}
	$v \gets $ root of Tree\;
	\While{$v \neq \Lambda$ }{
		\If{Key = $\keyit(v)$}{
			\textbf{halt}\;
		}
		\eIf{Key $> \keyit(v)$}{
			$v \gets \rightit(v)$\;
		}{
			$v \gets \leftit(v)$\;
		}
	}
	\caption{Find(Tree, Key)}
\end{algorithm}
